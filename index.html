​ 晚上回家后直接原神启动到12点， 碎了。

Day 1:

​ 起的很早， 然后原神启动。。。。。。我好颓中午早早地吃完了饭（我要说， 猪蹄好好吃！）， 来到了七中门口。围着七中逛了几圈， 就进场了。那个守门的老师太水了， 我带着智能手表居然进去了。要不是我正直害怕被误杀， 主动交了，o(￣ヘ￣o＃)。我背后那哥们儿直接开始动电脑， 被警告力（乐）。

​ 很快就发题了。这七中什么破键盘啊， 手感极差， 又没有隔板， 不敢东张西望。 旁边那哥们一来就搞心态， 直接狂敲键盘， 敲得还贼大声。不管他， 静下心来， 认真看题。

​ 先看T1， 好， 签到题， 直接暴搜解决。十多分钟写完了， 跑大样例， 过了。再看T2T3T4， 都想不出来正解， 那就先睡一会儿， 互搞心态。睡了大概两个小时， 起来匆匆打了三道题的特殊性质， 连Linux都没测就交了。

​ 算了一下， 预估 100 + 50 + 50 + 50， 还行吧， 感觉能拿一等。

​ 回到家， 整个群里面都在原神启动， 那我也浅浅启动一下吧。

Day 2:

​ 崩溃了。

​ 上小图灵一测， 第二题只拿了30分， 第三题暴力输入打挂了， 第四题不知道为什么一分没拿到， 一下变成了 100 + 30 + 0 + 0， 瞬间掉成了二等奖。崩溃了！！！！！早知道就不睡觉了，检查一遍我的暴力都不会这样。唉， 把自己的一等奖拱手相让了属于是。等着看看最后成绩还有没有翻的可能吧， 毕竟要相信CCF的脚造数据。

#include <bits/stdc++.h>
using namespace std;

#define int long long
#define LL long long
const int N = 1010;
const int M = N * N / 2;
const int MOD = 1e9 + 7;
template <typename T> inline void read(T &x){
    int w = 1;char s;x = 0;
    while (!isdigit(s = getchar())) if (s == '-') w = -1;
    while (isdigit(s)) x = (x << 3) + (x << 1), x += s - '0', s = getchar();
    x *= w;
}

int n, m, T;
bitset<1010>ver[N];
vector <int> may1, may2;
inline void add(int u, int v){
    ver[u] ^= (1 << (v - 1));
}

inline int ksc(int x, int y){
    int ans = 0;int temp = x;
    while (y){
        if (y & 1) ans += temp;
        y >>= 1;
        temp <<= 1;
        ans %= MOD;
    }
    return ans;
}
signed main(){
    freopen("trz.in","r",stdin);
    freopen("trz.out","w",stdout);
    read(T);
    while (T--){

        may1.clear();may2.clear();
        int ans = 0;
        read(n), read(m);
        for (register int i = 1; i <= n; i++)
            ver[i].reset();
        for (register int i = 1, u, v; i <= m; read(u), read(v), add(u, v), add(v, u), i++);
        for (register int i = 1; i <= n; i++){
            if (ver[i].count() >= 6) may1.push_back(i);
            if (ver[i].count() >= 4) may2.push_back(i);
        }
        for (register auto i : may1){
            for (register auto j : may2){
                if (i != j){
                    ver[0]=ver[i]&ver[j];
                    register int cnt = ver[i].count(), cnt2 = ver[0].count();
                    if(ver[i][j - 1])  cnt--;
                    if (cnt2 >= 4){
                        ans = (ans + ksc(ksc(cnt2, ksc(cnt2 - 1, ksc(cnt2 - 2, cnt2 - 3))) / 24,((ksc(cnt - 4, cnt - 5))) / 2ll)) % MOD;
                    }
                }
            }
        }
        printf("%lld\n", ans);
    }
}
