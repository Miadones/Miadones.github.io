怎么都调不对， 求调

#include <bits/stdc++.h>
using namespace std;

#define int long long
#define LL long long
const int N = 1010;
const int M = N * N / 2;
const int MOD = 1e9 + 7;
template <typename T> inline void read(T &x){
    int w = 1;char s;x = 0;
    while (!isdigit(s = getchar())) if (s == '-') w = -1;
    while (isdigit(s)) x = (x << 3) + (x << 1), x += s - '0', s = getchar();
    x *= w;
}

int n, m, T;
bitset<1010>ver[N];
vector <int> may1, may2;
inline void add(int u, int v){
    ver[u] ^= (1 << (v - 1));
}

inline int ksc(int x, int y){
    int ans = 0;int temp = x;
    while (y){
        if (y & 1) ans += temp;
        y >>= 1;
        temp <<= 1;
        ans %= MOD;
    }
    return ans;
}
signed main(){
    freopen("trz.in","r",stdin);
    freopen("trz.out","w",stdout);
    read(T);
    while (T--){

        may1.clear();may2.clear();
        int ans = 0;
        read(n), read(m);
        for (register int i = 1; i <= n; i++)
            ver[i].reset();
        for (register int i = 1, u, v; i <= m; read(u), read(v), add(u, v), add(v, u), i++);
        for (register int i = 1; i <= n; i++){
            if (ver[i].count() >= 6) may1.push_back(i);
            if (ver[i].count() >= 4) may2.push_back(i);
        }
        for (register auto i : may1){
            for (register auto j : may2){
                if (i != j){
                    ver[0]=ver[i]&ver[j];
                    register int cnt = ver[i].count(), cnt2 = ver[0].count();
                    if(ver[i][j - 1])  cnt--;
                    if (cnt2 >= 4){
                        ans = (ans + ksc(ksc(cnt2, ksc(cnt2 - 1, ksc(cnt2 - 2, cnt2 - 3))) / 24,((ksc(cnt - 4, cnt - 5))) / 2ll)) % MOD;
                    }
                }
            }
        }
        printf("%lld\n", ans);
    }
}
