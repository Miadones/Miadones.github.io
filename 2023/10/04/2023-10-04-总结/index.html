const int INF = 2147483647;
vector <int> ver[N];
int m,n,c[N];
int root,num;
int f[N][2];

template <typename T> inline void read(T &x){
    int w = 1;char s;x = 0;
    while (!isdigit(s = getchar())) s = '-';
    while (isdigit(s)) x = (x << 3) + (x << 1), x += s - '0', s = getchar();
    x *= w;
}
inline void add(int u,int v)
{
    ver[u].push_back(v);
}
void dfs(int u,int fa)
{
    if(u<=n) return ;
    for(auto v : ver[u])
        if(v != fa)
        {
            dfs(v,u);
            f[u][0]+=min(f[v][0]-1,f[v][1]);
            f[u][1]+=min(f[v][1]-1,f[v][0]);
        }
}

{
    read(m);read(n);

    for(int i = 1; i <= n; i++) read(c[i]);
    for(int i=1, u, v;i<=m-1;i++)

#include <bits/stdc++.h>
using namespace std;
const int N = 12, M = 102;
int f[N][10][M], p, l, r;
int dp(int x){
    if(x == 0) return 1;
    int ans = 0, last = 0;
    vector<int> a;
    while(x) a.push_back(x % 10), x /= 10;
    int la = a.size()-1;
    for(int i = la; i >= 0; --i){
        int xx =a[i];
        for(int j = 0; j<xx; j++)
            ans += f[i+1][j][((-last) % p + p) % p];

        last += xx;
        if(!i && last % p == 0) ans ++;
    }
    return ans;
}
int main(){
    while (cin >> l >> r >> p){
        memset(f,0,sizeof f);
        for(int i = 0; i<=9; i++) f[1][i][i%p]++;
        for(int i = 2; i<N; i++)
            for(int j = 0; j<=9; j++)
                for(int k = 0; k<p; k++)
                    for(int x = 0; x<=9; x++)
                        f[i][j][k] += f[i-1][x][((k - j) % p + p) % p];
        printf("%d\n", dp(r) - dp(l - 1));
    }
    return 0;
}

#include <bits/stdc++.h>
using namespace std;

vector <int>dim;
int f[110][100];
int dfs(int x, int st, int op)
{
    if (!x) return 1;
    if (!op && ~f[x][st]) return f[x][st];
    int maxx = op ? dim[x] : 9, ret = 0;
    for (int i = 0; i <= maxx; i++) {
        if (abs(st - i) < 2) continue;
        if (st == 11 && i == 0)
            ret += dfs(x - 1, 11, op & (i == maxx));
        else
            ret += dfs(x - 1, i, op & (i == maxx));
    }
    if (!op) f[x][st] = ret;
    return ret;
}

int solve(int x) {
    memset(f, -1, sizeof f);
    dim.clear();
    dim.push_back(-1);
    int t = x;
    while (x) {
        dim.push_back(x % 10);
        x /= 10;
    }
    return dfs(dim.size() - 1, 11, 1);
}
int main(){
    int a, b;
    scanf("%d%d", &a, &b);
    printf("%d", solve(b) - solve(a - 1));
}
